        // è®¾ç½®ç”»å¸ƒå¤§å°

class EasingAnimation {
    constructor(canvas) {
        this.canvas = canvas;
        this.ctx = canvas.getContext('2d');
        this.x = 0;
        this.targetX = 0;
        this.init();
    }

    init() {
        this.resize();
        window.addEventListener('resize', () => this.resize());
        this.canvas.addEventListener('click', (e) => {
            this.targetX = e.clientX;
        });
        this.animate();
    }

    resize() {
        this.canvas.width = window.innerWidth;
        this.canvas.height = window.innerHeight;
    }

    // ç¼“åŠ¨å‡½æ•°
    easeOutQuad(t) {
        return t * (2 - t);
    }

    animate() {
        this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
        
        // è®¡ç®—ç¼“åŠ¨
        const dx = this.targetX - this.x;
        this.x += dx * 0.1;
        
        // ç»˜åˆ¶
        this.ctx.fillStyle = 'blue';
        this.ctx.beginPath();
        this.ctx.arc(this.x, this.canvas.height / 2, 20, 0, Math.PI * 2);
        this.ctx.fill();
        
        requestAnimationFrame(() => this.animate());
    }
}

class ParticleSystem {
    constructor(canvas) {
        this.canvas = canvas;
        this.ctx = canvas.getContext('2d');
        this.particles = [];
        this.init();
    }

    init() {
        this.resize();
        window.addEventListener('resize', () => this.resize());
        this.createParticles();
        this.animate();
    }

    resize() {
        this.canvas.width = window.innerWidth;
        this.canvas.height = window.innerHeight;
    }

    createParticles() {
        for (let i = 0; i < 100; i++) {
            this.particles.push({
                x: Math.random() * this.canvas.width,
                y: Math.random() * this.canvas.height,
                size: Math.random() * 5 + 1,
                speedX: Math.random() * 3 - 1.5,
                speedY: Math.random() * 3 - 1.5,
                color: `hsl(${Math.random() * 360}, 50%, 50%)`
            });
        }
    }

    animate() {
        this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
        
        this.particles.forEach(particle => {
            // æ›´æ–°ä½ç½®
            particle.x += particle.speedX;
            particle.y += particle.speedY;
            
            // è¾¹ç•Œæ£€æŸ¥
            if (particle.x < 0 || particle.x > this.canvas.width) {
                particle.speedX *= -1;
            }
            if (particle.y < 0 || particle.y > this.canvas.height) {
                particle.speedY *= -1;
            }
            
            // ç»˜åˆ¶ç²’å­
            this.ctx.beginPath();
            this.ctx.arc(particle.x, particle.y, particle.size, 0, Math.PI * 2);
            this.ctx.fillStyle = particle.color;
            this.ctx.fill();
        });
        
        requestAnimationFrame(() => this.animate());
    }
}

class PhysicsAnimation {
    constructor(canvas) {
        this.canvas = canvas;
        this.ctx = canvas.getContext('2d');
        this.balls = [];
        this.gravity = 0.5;
        this.friction = 0.99;
        this.init();
    }

    init() {
        this.resize();
        window.addEventListener('resize', () => this.resize());
        this.createBalls();
        this.animate();
    }

    resize() {
        this.canvas.width = window.innerWidth;
        this.canvas.height = window.innerHeight;
    }

    createBalls() {
        for (let i = 0; i < 10; i++) {
            this.balls.push({
                x: Math.random() * this.canvas.width,
                y: Math.random() * this.canvas.height / 2,
                radius: Math.random() * 20 + 10,
                speedX: Math.random() * 4 - 2,
                speedY: 0,
                color: `hsl(${Math.random() * 360}, 50%, 50%)`
            });
        }
    }

    animate() {
        this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
        
        this.balls.forEach(ball => {
            // åº”ç”¨é‡åŠ›
            ball.speedY += this.gravity;
            
            // æ›´æ–°ä½ç½®
            ball.x += ball.speedX;
            ball.y += ball.speedY;
            
            // åº”ç”¨æ‘©æ“¦åŠ›
            ball.speedX *= this.friction;
            ball.speedY *= this.friction;
            
            // ç¢°æ’æ£€æµ‹
            if (ball.x + ball.radius > this.canvas.width) {
                ball.x = this.canvas.width - ball.radius;
                ball.speedX *= -0.8;
            }
            if (ball.x - ball.radius < 0) {
                ball.x = ball.radius;
                ball.speedX *= -0.8;
            }
            if (ball.y + ball.radius > this.canvas.height) {
                ball.y = this.canvas.height - ball.radius;
                ball.speedY *= -0.8;
            }
            if (ball.y - ball.radius < 0) {
                ball.y = ball.radius;
                ball.speedY *= -0.8;
            }
            
            // ç»˜åˆ¶çƒ
            this.ctx.beginPath();
            this.ctx.arc(ball.x, ball.y, ball.radius, 0, Math.PI * 2);
            this.ctx.fillStyle = ball.color;
            this.ctx.fill();
        });
        
        requestAnimationFrame(() => this.animate());
    }
}

class MouseFollow {
    constructor(canvas) {
        this.canvas = canvas;
        this.ctx = canvas.getContext('2d');
        this.particles = [];
        this.mouse = { x: 0, y: 0 };
        this.init();
    }

    init() {
        this.resize();
        window.addEventListener('resize', () => this.resize());
        this.canvas.addEventListener('mousemove', (e) => {
            const rect = this.canvas.getBoundingClientRect();
            this.mouse.x = e.clientX - rect.left;
            this.mouse.y = e.clientY - rect.top;
        });
        this.createParticles();
        this.animate();
    }

    resize() {
        this.canvas.width = window.innerWidth;
        this.canvas.height = window.innerHeight;
    }

    createParticles() {
        for (let i = 0; i < 50; i++) {
            this.particles.push({
                x: Math.random() * this.canvas.width,
                y: Math.random() * this.canvas.height,
                size: Math.random() * 5 + 2,
                speedX: 0,
                speedY: 0,
                color: `hsl(${Math.random() * 360}, 50%, 50%)`
            });
        }
    }

    animate() {
        this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
        
        this.particles.forEach(particle => {
            // è®¡ç®—åˆ°é¼ æ ‡çš„è·ç¦»
            const dx = this.mouse.x - particle.x;
            const dy = this.mouse.y - particle.y;
            const distance = Math.sqrt(dx * dx + dy * dy);
            
            // æ ¹æ®è·ç¦»è®¡ç®—é€Ÿåº¦
            if (distance < 100) {
                const angle = Math.atan2(dy, dx);
                const force = (100 - distance) / 100;
                
                particle.speedX -= Math.cos(angle) * force * 0.2;
                particle.speedY -= Math.sin(angle) * force * 0.2;
            }
            
            // æ›´æ–°ä½ç½®
            particle.x += particle.speedX;
            particle.y += particle.speedY;
            
            // åº”ç”¨æ‘©æ“¦åŠ›
            particle.speedX *= 0.95;
            particle.speedY *= 0.95;
            
            // ç»˜åˆ¶ç²’å­
            this.ctx.beginPath();
            this.ctx.arc(particle.x, particle.y, particle.size, 0, Math.PI * 2);
            this.ctx.fillStyle = particle.color;
            this.ctx.fill();
        });
        
        requestAnimationFrame(() => this.animate());
    }
}

class ObjectPool {
    constructor(createFn, resetFn, initialSize = 100) {
        this.createFn = createFn;
        this.resetFn = resetFn;
        this.pool = [];
        this.active = new Set();
        
        // åˆå§‹åŒ–å¯¹è±¡æ± 
        for (let i = 0; i < initialSize; i++) {
            this.pool.push(this.createFn());
        }
    }

    // è·å–å¯¹è±¡
    get() {
        let obj = this.pool.pop();
        if (!obj) {
            obj = this.createFn();
        }
        this.active.add(obj);
        return obj;
    }

    // é‡Šæ”¾å¯¹è±¡
    release(obj) {
        if (this.active.has(obj)) {
            this.resetFn(obj);
            this.active.delete(obj);
            this.pool.push(obj);
        }
    }

    // é‡Šæ”¾æ‰€æœ‰å¯¹è±¡
    releaseAll() {
        this.active.forEach(obj => {
            this.resetFn(obj);
            this.pool.push(obj);
        });
        this.active.clear();
    }
}

class FrameRateController {
    constructor(targetFPS = 60) {
        this.targetFPS = targetFPS;
        this.frameInterval = 1000 / targetFPS;
        this.lastFrameTime = 0;
    }

    shouldRender(currentTime) {
        if (currentTime - this.lastFrameTime >= this.frameInterval) {
            this.lastFrameTime = currentTime;
            return true;
        }
        return false;
    }
}

// ä½¿ç”¨ç¤ºä¾‹

function animate(currentTime) {
    if (fpsController.shouldRender(currentTime)) {
        // æ‰§è¡ŒåŠ¨ç”»æ›´æ–°å’Œç»˜åˆ¶
        update();
        draw();
    }
    requestAnimationFrame(animate);
}

    class Animation {
    constructor() {
        this.isRunning = false;
        this.animationId = null;
    }

    start() {
        if (!this.isRunning) {
            this.isRunning = true;
            this.animate();
        }
    }

    stop() {
        if (this.isRunning) {
            this.isRunning = false;
            cancelAnimationFrame(this.animationId);
        }
    }

    animate() {
        if (!this.isRunning) return;
        
        // åŠ¨ç”»é€»è¾‘
        this.update();
        this.draw();
        
        this.animationId = requestAnimationFrame(() => this.animate());
    }
}
